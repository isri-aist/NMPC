142  const VarDimVector & g,
143  const VarDimVector & lower,
144  const VarDimVector & upper,
145  const VarDimVector & initial_x)
146  {
147  // Initialize objective value
148  VarDimVector x = initial_x.cwiseMin(upper).cwiseMax(lower);
149  double obj = x.dot(g) + 0.5 * x.dot(H * x);
150  double old_obj = obj;
151 
152  // Initialize trace data
153  trace_data_list_.clear();

156  initial_trace_data.x = x;
157  initial_trace_data.obj = obj;
158  trace_data_list_.push_back(initial_trace_data);
159 
160  // Main loop
161  int retval = 0;
162  int factorization_num = 0;
163  VarDimVector grad = VarDimVector::Zero(var_dim_);
164  VarDimArray clamped_flag = VarDimArray::Zero(var_dim_);
165  VarDimArray old_clamped_flag = clamped_flag;
166  int iter = 1;
167  for(;; iter++)
168  {
169  // Append trace data
170  trace_data_list_.push_back(TraceData(var_dim_));
171  auto & trace_data = trace_data_list_.back();
172  trace_data.iter = iter;
173 
174  // Check relative improvement
175  if(iter > 1 && (old_obj - obj) < config_.rel_improve_thre * std::abs(old_obj))
176  {
177  retval = 4;
178  break;
179  }
180  old_obj = obj;
181 
182  // Calculate gradient
183  grad = g + H * x;
184 
185  // Find clamped dimensions
186  old_clamped_flag = clamped_flag;
187  clamped_flag.setConstant(false);
188  clamped_flag =
189  ((x.array() == lower.array() && grad.array() > 0) || (x.array() == upper.array() && grad.array() < 0))
190  .select(true, clamped_flag);
191 
192  // Set clamped and free indices
193  std::vector<int> clamped_idxs;
194  std::vector<int> free_idxs;
195  for(int i = 0; i < clamped_flag.size(); i++)
196  {
197  if(clamped_flag[i])
198  {
199  clamped_idxs.push_back(i);
200  }
201  else
202  {
203  free_idxs.push_back(i);
204  }
205  }
206 
207  // Check for all clamped
208  if(clamped_flag.all())
209  {
210  retval = 6;
211  break;
212  }
213 
214  // Factorize if clamped has changed
215  if(iter == 1 || (clamped_flag != old_clamped_flag).any())
216  {
217  // Set H_free
218  Eigen::MatrixXd H_free(free_idxs.size(), free_idxs.size());
219  for(int i = 0; i < free_idxs.size(); i++)
220  {
221  for(int j = 0; j < free_idxs.size(); j++)
222  {
223  H_free(i, j) = H(free_idxs[i], free_idxs[j]);
224  }
225  }
226 
227  // Cholesky decomposition
228  llt_ = std::make_unique<Eigen::LLT<Eigen::MatrixXd>>(H_free);
229  if(llt_->info() == Eigen::NumericalIssue)
230  {

232  {
233  std::cout << "[BoxQP] H_free is not positive definite in Cholesky decomposition (LLT)." << std::endl;
234  }
235  retval = -1;
236  break;
237  }
238 
239  factorization_num++;
240  }
241 
242  // Check gradient norm
243  double grad_norm = 0;
244  for(int i = 0; i < free_idxs.size(); i++)
245  {
246  grad_norm += std::pow(grad[free_idxs[i]], 2);
247  }
248  if(grad_norm < std::pow(config_.grad_thre, 2))
249  {
250  retval = 5;
251  break;
252  }
253 
254  // Calculate search direction
255  Eigen::VectorXd x_clamped(clamped_idxs.size());
256  Eigen::VectorXd x_free(free_idxs.size());
257  Eigen::VectorXd g_free(free_idxs.size());
258  Eigen::MatrixXd H_free_clamped(free_idxs.size(), clamped_idxs.size());
259  for(int i = 0; i < clamped_idxs.size(); i++)
260  {
261  x_clamped[i] = x[clamped_idxs[i]];
262  }
263  for(int i = 0; i < free_idxs.size(); i++)
264  {
265  x_free[i] = x[free_idxs[i]];
266  g_free[i] = g[free_idxs[i]];
267  for(int j = 0; j < clamped_idxs.size(); j++)
268  {
269  H_free_clamped(i, j) = H(free_idxs[i], clamped_idxs[j]);
270  }
271  }
272  Eigen::VectorXd grad_free_clamped = g_free + H_free_clamped * x_clamped;
273  Eigen::VectorXd search_dir_free = -1 * llt_->solve(grad_free_clamped) - x_free;
274  VarDimVector search_dir = VarDimVector::Zero(var_dim_);
275  for(int i = 0; i < free_idxs.size(); i++)
276  {
277  search_dir[free_idxs[i]] = search_dir_free[i];
278  }
279 
280  // Check for descent direction
281  double search_dir_grad = search_dir.dot(grad);
282  if(search_dir_grad > 0) // This should not happen
283  {
284  if(config_.print_level >= 1)
285  {
286  std::cout << "[BoxQP] search_dir_grad is negative: " << search_dir_grad << std::endl;
287  }
288  retval = -2;
289  break;
290  }
291 
292  // Armijo linesearch
293  double step = 1;
294  int step_num = 0;
295  VarDimVector x_candidate = (x + step * search_dir).cwiseMin(upper).cwiseMax(lower);
296  double obj_candidate = x_candidate.dot(g) + 0.5 * x_candidate.dot(H * x_candidate);
297  while((obj_candidate - old_obj) / (step * search_dir_grad) < config_.armijo_param)
298  {
299  step = step * config_.step_factor;
300  step_num++;
301  x_candidate = (x + step * search_dir).cwiseMin(upper).cwiseMax(lower);
302  obj_candidate = x_candidate.dot(g) + 0.5 * x_candidate.dot(H * x_candidate);
303  if(step < config_.min_step)
304  {
305  retval = 2;
306  break;
307  }
308  }
309 
310  // Print
311  if(config_.print_level >= 3)
312  {
313  std::cout << "[BoxQP] iter: " << iter << ", obj: " << obj << ", grad_norm: " << grad_norm
314  << ", obj_update, : " << old_obj - obj_candidate << ", step: " << step
315  << ", clamped_flag_num: " << clamped_idxs.size() << std::endl;
316  }
317 
318  // Set trace data
319  trace_data.x = x;
320  trace_data.obj = obj;
321  trace_data.search_dir = search_dir;
322  trace_data.clamped_flag = clamped_flag;
323  trace_data.factorization_num = factorization_num;
324  trace_data.step_num = step_num;
325 
326  // Accept candidate
327  x = x_candidate;
328  obj = obj_candidate;
329 
330  // Check loop termination
331  if(iter == config_.max_iter)
332  {
333  retval = 1;
334  break;
335  }
336  }
337 
338  // Print
339  std::unordered_map<int, std::string> retstr = {{-2, "Gradient of search direction is negative"},
340  {-1, "Hessian is not positive definite"},
341  {0, "No descent direction found"},
342  {1, "Maximum main iterations exceeded"},
343  {2, "Maximum line-search iterations exceeded"},
344  {3, "No bounds, returning Newton point"},
345  {4, "Improvement smaller than tolerance"},
346  {5, "Gradient norm smaller than tolerance"},
347  {6, "All dimensions are clamped"}};
348

353  << ", factorization_num: " << factorization_num << std::endl;
